## 📊 论文统计

```dataviewjs
const papers = dv.pages('"📚 论文库/论文笔记"');
const total = papers.length;
const unread = papers.where(p => p.status === "unread").length;
const reading = papers.where(p => p.status === "reading").length;
const finished = papers.where(p => p.status === "finished").length;

dv.table(
    ["指标", "数量"],
    [
        ["📄 论文笔记总数", total],
        ["⬜ 未读", unread],
        ["🟡 阅读中", reading],
        ["✅ 已读", finished],
    ]
);
```

## 📚 论文列表

```dataviewjs
dv.table(
    ["论文", "年份", "状态"],
    dv.pages('"📚 论文库/论文笔记"')
        .sort(p => p.file.ctime, 'desc')
        .map(p => [p.file.link, p.year || "-", p.status || "unread"])
);
```

## 📁 PDF库

```dataviewjs
const pdfFolder = app.vault.getAbstractFileByPath("📚 论文库/PDF库");
if (pdfFolder && pdfFolder.children) {
    const pdfFiles = pdfFolder.children.filter(f => f.extension === "pdf");
    dv.paragraph(`📁 共有 **${pdfFiles.length}** 个PDF文件`);
    
    dv.table(
        ["文件名", "操作"],
        pdfFiles.slice(0, 10).map(f => [f.name, "[打开](" + f.path + ")"])
    );
} else {
    dv.paragraph("📁 PDF库暂无文件");
}
```

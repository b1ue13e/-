async function openNoteTemplate() {
    const templatePath = '🧠 学习系统/📋 笔记模板-带完整Frontmatter.md';
    const file = app.vault.getAbstractFileByPath(templatePath);
    if (file) {
        await app.workspace.getLeaf(false).openFile(file);
        new Notice('✅ 已打开笔记模板');
    } else {
        new Notice('❌ 模板文件未找到: ' + templatePath);
    }
}

module.exports = openNoteTemplate;

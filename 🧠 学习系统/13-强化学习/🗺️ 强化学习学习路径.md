```dataviewjs
dv.table(
    ["笔记", "阶段/维度", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/13-强化学习"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let category = "-";
            if (p.file.path.includes("01-")) category = "🌱 基础";
            else if (p.file.path.includes("02-")) category = "🎯 MDP";
            else if (p.file.path.includes("03-")) category = "💰 值函数";
            else if (p.file.path.includes("04-")) category = "🧠 Q-Learning";
            else if (p.file.path.includes("05-")) category = "📈 策略梯度";
            else if (p.file.path.includes("06-")) category = "🎭 A-C";
            else if (p.file.path.includes("07-")) category = "🚀 PPO";
            else if (p.file.path.includes("知识地图")) category = "🗺️ 地图";
            return [p.file.link, category, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

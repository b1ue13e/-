## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段/维度", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/10-数据结构与算法"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let category = "-";
            if (p.file.path.includes("01-")) category = "🌱 入门";
            else if (p.file.path.includes("02-")) category = "📚 线性";
            else if (p.file.path.includes("03-")) category = "🌲 树堆";
            else if (p.file.path.includes("04-")) category = "🕸️ 图论";
            else if (p.file.path.includes("05-")) category = "💡 进阶";
            else if (p.file.path.includes("06-")) category = "🎯 实战";
            else if (p.file.path.includes("知识地图")) category = "🗺️ 地图";
            return [p.file.link, category, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

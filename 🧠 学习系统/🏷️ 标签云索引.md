## ☁️ 全库标签概览

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name);
const normalizeTag = tag => String(tag || "").trim().replace(/^#/, "");
const pages = [
    ...dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array(),
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];
const counts = {};
pages.forEach(p => {
    toList(p.tags).map(normalizeTag).filter(Boolean).filter(tag => !tag.includes("模板")).forEach(tag => {
        counts[tag] = (counts[tag] || 0) + 1;
    });
});

dv.table(
    ["标签", "次数", "搜索"],
    Object.entries(counts)
        .sort((a, b) => b[1] - a[1] || a[0].localeCompare(b[0], "zh-Hans-CN"))
        .slice(0, 35)
        .map(([tag, count]) => [tag, count, `[搜索](obsidian://search?query=${encodeURIComponent("tag:#" + tag)})`])
);
```

## 🧹 标签治理提醒

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const normalizeTag = tag => String(tag || "").trim().replace(/^#/, "");
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const pages = [
    ...dv.pages('"🧠 学习系统"').where(p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name)).array(),
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];
const tagToPages = {};
pages.forEach(p => {
    toList(p.tags).map(normalizeTag).filter(Boolean).forEach(tag => {
        if (!tagToPages[tag]) tagToPages[tag] = [];
        tagToPages[tag].push(p);
    });
});
const singletonTags = Object.entries(tagToPages)
    .filter(([tag, list]) => list.length === 1 && !["学习", "论文", "书籍", "阅读"].includes(tag))
    .sort((a, b) => a[0].localeCompare(b[0], "zh-Hans-CN"));
const broadTags = ["学习", "论文", "书籍", "阅读", "索引", "主页"]
    .filter(tag => tagToPages[tag])
    .map(tag => [tag, tagToPages[tag].length, "只保留作类型入口，不承担主题分类"]);
const missingTags = pages.filter(p => toList(p.tags).length === 0).slice(0, 20);

dv.table(
    ["治理项", "数量", "建议"],
    [
        ["一次性标签", singletonTags.length, "优先判断是否可并入上位标签或 concept 字段"],
        ["过泛标签", broadTags.length, "类型标签可保留，但不要作为唯一标签"],
        ["缺少 tags 的笔记", missingTags.length, "新建或补齐类型标签 + 领域标签"]
    ]
);

if (broadTags.length > 0) {
    dv.header(3, "过泛标签");
    dv.table(["标签", "覆盖笔记", "处理建议"], broadTags);
}

if (singletonTags.length > 0) {
    dv.header(3, "一次性标签 Top 30");
    dv.table(
        ["标签", "所在笔记"],
        singletonTags.slice(0, 30).map(([tag, list]) => [tag, list[0].file.link])
    );
}
```

## 🏷️ 推荐标签结构

### 学习笔记
```yaml
tags: [学习, 机器学习, 优化算法]
```

### 论文笔记
```yaml
tags: [论文, 时间序列, 风电预测]
```

### 读书笔记
```yaml
tags: [书籍, 阅读, 方法论]
```

### 建议规则

- 第一层：内容类型，如 `学习`、`论文`、`书籍`、`项目`。
- 第二层：领域，如 `机器学习`、`量化金融`、`时间序列`。
- 第三层：方法或对象，如 `优化算法`、`回归诊断`、`GNN`。
- 具体知识点优先写进 `concept`，不要把标签膨胀成概念列表。

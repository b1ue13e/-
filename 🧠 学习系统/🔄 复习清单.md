# 🔄 复习清单

> 基于 `mastery / last_reviewed / next_review` 和题库错题记录安排复习，而不是只看创建时间。

---

## 📊 复习盘面

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const isTemplate = p =>
    p.file.name.includes("模板") ||
    (Array.isArray(p.tags) && p.tags.includes("模板"));
const toArray = value => Array.isArray(value) ? value : (value ? [value] : []);
const toPath = value => {
    if (!value) return "";
    if (typeof value === "string") return value.replace(/\.md$/, "");
    if (value.path) return value.path.replace(/\.md$/, "");
    return String(value).replace(/\.md$/, "");
};
const toTs = value => {
    if (!value) return null;
    if (typeof value === "number") return value;
    if (value.ts) return value.ts;
    const parsed = new Date(value);
    return Number.isNaN(parsed.getTime()) ? null : parsed.getTime();
};
const getMastery = page => {
    const parsed = Number(page.mastery);
    return Number.isFinite(parsed) && parsed > 0 ? parsed : 1;
};
const defaultInterval = mastery => ({1: 1, 2: 3, 3: 7, 4: 14, 5: 30}[mastery] || 7);
const getInterval = page => {
    const parsed = Number(page.review_interval);
    if (Number.isFinite(parsed) && parsed > 0) return parsed;
    return defaultInterval(getMastery(page));
};
const getNextReviewTs = page => {
    const explicit = toTs(page.next_review);
    if (explicit) return explicit;
    const lastReviewed = toTs(page.last_reviewed) || page.file.mtime.ts || page.file.ctime.ts;
    return lastReviewed + getInterval(page) * 86400000;
};

const questionPages = dv.pages('"🧠 学习系统/18-题库学习"')
    .where(p => p.type === "question" && !isTemplate(p))
    .array();
const studyPages = dv.pages('"🧠 学习系统"')
    .where(p =>
        p.type === "study" &&
        !excludedStudyDirs.some(dir => p.file.path.startsWith(dir))
    )
    .array();

const linkedQuestionCount = page => questionPages.filter(q =>
    toArray(q.linked_notes).map(toPath).includes(page.file.path.replace(/\.md$/, ""))
).length;
const weakQuestionCount = page => questionPages.filter(q =>
    toArray(q.linked_notes).map(toPath).includes(page.file.path.replace(/\.md$/, "")) &&
    Number(q.mistake_count || 0) > 0
).length;
const dueToday = studyPages.filter(p =>
    ["ongoing", "completed"].includes(p.status) &&
    getNextReviewTs(p) <= Date.now()
);

dv.table(
    ["指标", "数量"],
    [
        ["学习笔记总数", studyPages.length],
        ["今日应复习", dueToday.length],
        ["已挂题笔记", studyPages.filter(p => linkedQuestionCount(p) > 0).length],
        ["存在错题牵引的笔记", studyPages.filter(p => weakQuestionCount(p) > 0).length]
    ]
);
```

## 📌 今日应复习

    .filter(item => item.nextReviewTs <= Date.now())
    .sort((a, b) => a.mastery - b.mastery || a.nextReviewTs - b.nextReviewTs);

        dueToday.map(item => [
            item.page.file.link,
            item.mastery,
            item.linkedQuestionCount,
            item.weakQuestionCount,
            item.weakQuestionCount > 0

                : item.linkedQuestionCount > 0

        ])

## 🧱 待补元数据

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const notes = dv.pages('"🧠 学习系统"')
    .where(p =>
        p.type === "study" &&
        !excludedStudyDirs.some(dir => p.file.path.startsWith(dir))
    )
    .array();

const todo = notes
    .map(p => {
        const missing = [];
        if (!p.concept || p.concept.length === 0) missing.push("concept");
        if (!p.prerequisites || p.prerequisites.length === 0) missing.push("prerequisites");
        if (!p.related || p.related.length === 0) missing.push("related");
        if (!p.mastery) missing.push("mastery");
        if (!p.next_review) missing.push("next_review");
        return { page: p, missing };
    })
    .filter(item => item.missing.length > 0)
    .sort((a, b) => b.missing.length - a.missing.length)
    .slice(0, 15);

if (todo.length === 0) {
    dv.paragraph("✅ 当前学习笔记的关键元数据已经补齐。");
} else {
    dv.table(
        ["笔记", "缺少字段", "最小补全动作"],
        todo.map(item => [
            item.page.file.link,
            item.missing.join(" / "),
            item.missing.includes("concept") ? "先补 1-3 个 concept" : "补上掌握度与下次复习日期"
        ])
    );
}
```

## 🔓 可推进的 locked 主题

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const notes = dv.pages('"🧠 学习系统"')
    .where(p =>
        p.type === "study" &&
        !excludedStudyDirs.some(dir => p.file.path.startsWith(dir))
    )
    .array();
const activeNotes = notes.filter(p => ["ongoing", "completed"].includes(p.status));
const activeTopics = new Set(activeNotes.map(p => p.topic).filter(Boolean));
const activeConcepts = new Set(activeNotes.flatMap(p => p.concept || []).filter(Boolean));
const readyLocked = notes
    .filter(p => p.status === "locked" && Array.isArray(p.prerequisites) && p.prerequisites.length > 0)
    .filter(p => p.prerequisites.every(pr => activeTopics.has(pr) || activeConcepts.has(pr)))
    .slice(0, 10);

if (readyLocked.length === 0) {
    dv.paragraph("当前没有前置已满足但仍然锁定的主题。");
} else {
    dv.table(
        ["主题", "前置已满足", "建议动作"],
        readyLocked.map(p => [
            p.file.link,
            (p.prerequisites || []).join(" / "),
            "切换到 ongoing，并先补 1 道概念题"
        ])
    );
}
```

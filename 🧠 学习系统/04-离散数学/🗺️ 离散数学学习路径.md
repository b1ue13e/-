## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/04-离散数学"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let stage = "-";
            if (p.file.path.includes("01-")) stage = "📐 基础";
            else if (p.file.path.includes("02-")) stage = "🌐 深化";
            else if (p.file.path.includes("03-")) stage = "🚀 进阶";
            else if (p.file.path.includes("04-")) stage = "🎯 贯通";
            return [p.file.link, stage, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

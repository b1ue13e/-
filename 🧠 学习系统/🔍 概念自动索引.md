# 🔍 概念自动索引

> 这页用来回答：**一个概念在哪些学习笔记、论文笔记和读书笔记中反复出现？哪些资源里出现了概念但还没有形成原子学习卡？**

## 📊 概念覆盖看板

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const normalizeConcept = concept => String(concept || "").trim().replace(/^[^\u4e00-\u9fa5A-Za-z0-9]+/, "").replace(/\s+/g, " ");
const isUsefulConcept = concept => {
    const text = normalizeConcept(concept);
    return text && !/^返回/.test(text) && !/^模块\d+[:：]/.test(text) && text !== "和其他笔记的关系";
};
const hasUsefulConcept = p => toList(p.concept).some(isUsefulConcept);
const studyPages = dv.pages('"🧠 学习系统"').where(p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name)).array();
const paperPages = dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array();
const bookPages = dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array();
const pages = [...studyPages, ...paperPages, ...bookPages];
const concepts = pages.flatMap(p => toList(p.concept).filter(isUsefulConcept).map(normalizeConcept));
const uniqueConcepts = [...new Set(concepts)];

dv.table(
    ["指标", "数量"],
    [
        ["纳入统计的学习笔记", studyPages.length],
        ["纳入统计的论文笔记", paperPages.length],
        ["纳入统计的读书笔记", bookPages.length],
        ["已挂 concept 的笔记", pages.filter(hasUsefulConcept).length],
        ["独立概念数", uniqueConcepts.length],
        ["概念总出现次数", concepts.length]
    ]
);
```

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");

const isUsefulConcept = concept => {

};
const sourceLabel = p => p.file.path.startsWith("📚 书库") ? "📚 书籍" : (p.file.path.startsWith("📚 论文库") ? "📄 论文" : "🎓 学习");

    ...dv.pages('"🧠 学习系统"').where(p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name)).array(),
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];
const buckets = {};
pages.forEach(p => {
    toList(p.concept).filter(isUsefulConcept).map(normalizeConcept).forEach(concept => {
        if (!buckets[concept]) buckets[concept] = { count: 0, sources: new Set(), pages: [] };
        buckets[concept].count += 1;
        buckets[concept].sources.add(sourceLabel(p));
        buckets[concept].pages.push(p);
    });
});

dv.table(
    ["概念", "次数", "来源", "代表笔记"],
    Object.entries(buckets)
        .sort((a, b) => b[1].count - a[1].count || a[0].localeCompare(b[0], "zh-Hans-CN"))
        .slice(0, 30)
        .map(([concept, item]) => [
            `[[#${concept}|${concept}]]`,
            item.count,
            [...item.sources].join(" / "),
            item.pages.slice(0, 3).map(p => p.file.link)
        ])
);
```

## 🧩 资源概念待转卡

> 书籍或论文里出现、但学习系统中还没有承载卡的概念，适合优先转成原子学习笔记。

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const normalizeConcept = concept => String(concept || "").trim().replace(/^[^\u4e00-\u9fa5A-Za-z0-9]+/, "").replace(/\s+/g, " ");
const isUsefulConcept = concept => {
    const text = normalizeConcept(concept);
    return text && !/^返回/.test(text) && !/^模块\d+[:：]/.test(text) && text !== "和其他笔记的关系";
};
const studyPages = dv.pages('"🧠 学习系统"').where(p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name)).array();
const resourcePages = [
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];
const studyConcepts = new Set(studyPages.flatMap(p => toList(p.concept).filter(isUsefulConcept).map(normalizeConcept)));
const resourceConcepts = {};
resourcePages.forEach(p => {
    toList(p.concept).filter(isUsefulConcept).map(normalizeConcept).forEach(concept => {
        if (!resourceConcepts[concept]) resourceConcepts[concept] = [];
        resourceConcepts[concept].push(p);
    });
});
const rows = Object.entries(resourceConcepts)
    .filter(([concept]) => !studyConcepts.has(concept))
    .sort((a, b) => b[1].length - a[1].length || a[0].localeCompare(b[0], "zh-Hans-CN"))
    .slice(0, 25)
    .map(([concept, pages]) => [
        concept,
        pages.length,
        pages.slice(0, 4).map(p => p.file.link),
        "新建学习卡，并把该概念写入 topic / concept"
    ]);

if (rows.length > 0) {
    dv.table(["资源概念", "出现次数", "来源", "建议动作"], rows);
} else {
    dv.paragraph("✅ 当前书籍 / 论文中的概念都已能在学习系统 concept 中找到承载。");
}
```

## 🧹 概念治理提醒

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",

];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const normalizeConcept = concept => String(concept || "").trim().replace(/^[^\u4e00-\u9fa5A-Za-z0-9]+/, "").replace(/\s+/g, " ");
const isUsefulConcept = concept => {
    const text = normalizeConcept(concept);
    return text && !/^返回/.test(text) && !/^模块\d+[:：]/.test(text) && text !== "和其他笔记的关系";
};
const pages = [
    ...dv.pages('"🧠 学习系统"').where(p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name)).array(),
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];
const missing = pages.filter(p => toList(p.concept).filter(isUsefulConcept).length === 0);
const overloaded = pages.filter(p => toList(p.concept).filter(isUsefulConcept).length > 8);
const sentenceLike = [];
pages.forEach(p => toList(p.concept).filter(isUsefulConcept).map(normalizeConcept).forEach(concept => {
    if (concept.length > 24 || /[，。；、]/.test(concept)) sentenceLike.push([concept, p]);
}));

dv.table(
    ["治理项", "数量", "建议"],
    [
        ["缺少 concept 的笔记", missing.length, "至少补 2-5 个稳定概念"],
        ["concept 过多的笔记", overloaded.length, "拆成更小原子卡，或把部分概念移到 related"],
        ["疑似句子式概念", sentenceLike.length, "改成名词短语，方便聚合"]
    ]
);

if (missing.length > 0) {
    dv.header(3, "缺少 concept 的笔记示例");
    dv.table(["笔记", "类型", "状态"], missing.slice(0, 20).map(p => [p.file.link, p.type || "-", p.status || "-"]));
}
if (overloaded.length > 0) {
    dv.header(3, "concept 过多的笔记示例");
    dv.table(["笔记", "概念数", "概念"], overloaded.slice(0, 20).map(p => [p.file.link, toList(p.concept).length, toList(p.concept).slice(0, 8).join(" / ")]));
}
if (sentenceLike.length > 0) {
    dv.header(3, "疑似句子式概念示例");
    dv.table(["概念", "所在笔记"], sentenceLike.slice(0, 25).map(([concept, p]) => [concept, p.file.link]));
}
```

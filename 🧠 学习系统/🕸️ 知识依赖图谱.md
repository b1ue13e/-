## 📊 依赖总览

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const norm = value => String(value || "").trim().replace(/\s+/g, " ");
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");
const activeStatuses = new Set(["ongoing", "completed", "practicing"]);
const queuedStatuses = new Set(["locked", "planned", "unseen"]);
const notes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array();
const notesWithPrereqs = notes.filter(p => toList(p.prerequisites).map(norm).filter(Boolean).length > 0);
const uniquePrereqs = [...new Set(notesWithPrereqs.flatMap(p => toList(p.prerequisites).map(norm).filter(Boolean)))];
const topicSet = new Set(notes.map(p => norm(p.topic)).filter(Boolean));
const conceptSet = new Set(notes.flatMap(p => toList(p.concept).map(norm).filter(Boolean)));
const activeNotes = notes.filter(p => activeStatuses.has(p.status));
const activeTopics = new Set(activeNotes.map(p => norm(p.topic)).filter(Boolean));
const activeConcepts = new Set(activeNotes.flatMap(p => toList(p.concept).map(norm).filter(Boolean)));
const missingCarriers = uniquePrereqs.filter(pr => !topicSet.has(pr) && !conceptSet.has(pr));
const readyQueued = notes
    .filter(p => queuedStatuses.has(p.status) && toList(p.prerequisites).map(norm).filter(Boolean).length > 0)
    .filter(p => toList(p.prerequisites).map(norm).filter(Boolean).every(pr => activeTopics.has(pr) || activeConcepts.has(pr)));

dv.table(
    ["指标", "数量"],
    [
        ["纳入依赖分析的学习笔记", notes.length],
        ["带前置依赖的笔记", notesWithPrereqs.length],
        ["唯一前置知识节点", uniquePrereqs.length],
        ["已有承载笔记 / 概念的前置", uniquePrereqs.length - missingCarriers.length],
        ["真正缺少承载笔记的前置", missingCarriers.length],
        ["可直接启动的待学主题", readyQueued.length]
    ]
);
```

## 🚨 真正缺口

```dataviewjs

    "🧠 学习系统/20-PARA知识管理/",

];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const norm = value => String(value || "").trim().replace(/\s+/g, " ");
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");

const conceptSet = new Set(notes.flatMap(p => toList(p.concept).map(norm).filter(Boolean)));
const missingCarriers = [...new Set(notes.flatMap(p => toList(p.prerequisites).map(norm).filter(pr => pr && !topicSet.has(pr) && !conceptSet.has(pr))))];
const rows = missingCarriers
    .map(prereq => {
        const downstream = notes.filter(p => toList(p.prerequisites).map(norm).includes(prereq));
        return [
            prereq,
            downstream.length,
            [...new Set(downstream.map(p => p.file.path.split('/')[1]))].slice(0, 4).join(" / "),
            downstream.slice(0, 5).map(p => p.file.link),
            `[搜索](obsidian://search?query=${encodeURIComponent(prereq)})`
        ];
    })
    .sort((a, b) => b[1] - a[1] || a[0].localeCompare(b[0], "zh-Hans-CN"))
    .slice(0, 30);

if (rows.length > 0) {
    dv.table(["缺少承载的前置", "影响主题数", "主要路径", "受影响主题", "动作"], rows);
} else {
    dv.paragraph("✅ 当前所有 prerequisites 都至少能在现有笔记或 concept 中找到承载。");
}
```

## 🧹 依赖写法检查

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const norm = value => String(value || "").trim().replace(/\s+/g, " ");
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");
const notes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array();
const issueRows = [];
notes.forEach(p => {
    const prereqs = toList(p.prerequisites).map(norm).filter(Boolean);
    const duplicates = [...new Set(prereqs.filter((x, i) => prereqs.indexOf(x) !== i))];
    if (duplicates.length > 0) issueRows.push([p.file.link, "重复前置", duplicates.join(" / "), "去重"]);
    prereqs.filter(pr => pr.length > 28 || /[，。；、]/.test(pr)).slice(0, 3).forEach(pr => {
        issueRows.push([p.file.link, "疑似句子式前置", pr, "改成稳定 topic / concept"]);
    });
    prereqs.filter(pr => /^(基础|概念|导论|TODO|待补|无)$/i.test(pr)).forEach(pr => {
        issueRows.push([p.file.link, "过泛或占位前置", pr, "替换成具体前置知识"]);
    });
});

if (issueRows.length > 0) {
    dv.table(["笔记", "问题", "值", "建议"], issueRows.slice(0, 40));
} else {
    dv.paragraph("✅ 当前 prerequisites 暂未发现明显重复、过长或占位写法。");
}
```

## 🧠 当前依赖最重的主题

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const toList = value => {
    if (!value) return [];
    if (Array.isArray(value)) return value;
    if (typeof value.array === "function") return value.array();
    return [value];
};
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");
const notes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array();
const ranked = notes
    .map(p => ({
        page: p,
        prereqCount: toList(p.prerequisites).length,
        conceptCount: toList(p.concept).length,
        relatedCount: toList(p.related).length
    }))
    .filter(item => item.prereqCount + item.conceptCount + item.relatedCount > 0)
    .sort((a, b) => (b.prereqCount + b.conceptCount + b.relatedCount) - (a.prereqCount + a.conceptCount + a.relatedCount))
    .slice(0, 15);

dv.table(
    ["主题", "前置", "概念", "关联", "状态"],
    ranked.map(item => [
        item.page.file.link,
        item.prereqCount,
        item.conceptCount,
        item.relatedCount,
        item.page.status || "-"
    ])
);
```

## 💡 使用建议

- `prerequisites` 尽量写成已有 `topic` 或稳定 `concept`，避免写成句子。
- 高频前置如果没有承载卡，优先新建一张原子卡，而不是只在阶段页里解释。
- `related` 适合横向关联，`prerequisites` 只写“没有它就难以学习当前主题”的前置。
- `locked / planned / unseen` 表示故意排队；如果前置已满足，就应进入 `ongoing` 或拆成任务。

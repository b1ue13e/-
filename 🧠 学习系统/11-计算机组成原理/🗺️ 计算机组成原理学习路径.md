## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段/维度", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/11-计算机组成原理"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let category = "-";
            if (p.file.path.includes("00-")) category = "🗺️ 总览";
            else if (p.file.path.includes("01-")) category = "⚙️ CPU";
            else if (p.file.path.includes("02-")) category = "💾 内存";
            else if (p.file.path.includes("03-")) category = "🔄 调度";
            else if (p.file.path.includes("04-")) category = "🔒 并发";
            else if (p.file.path.includes("05-")) category = "✅ 复盘";
            else if (p.file.path.includes("知识地图")) category = "🗺️ 地图";
            return [p.file.link, category, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

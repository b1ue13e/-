## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段/维度", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/14-概率图模型"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let category = "-";
            if (p.file.path.includes("01-")) category = "📊 基础";
            else if (p.file.path.includes("02-")) category = "🔄 推断";
            else if (p.file.path.includes("03-")) category = "📚 学习";
            else if (p.file.path.includes("04-")) category = "🌐 贝叶斯";
            else if (p.file.path.includes("05-")) category = "🔗 HMM";
            else if (p.file.path.includes("06-")) category = "🚀 高级";
            else if (p.file.path.includes("知识地图")) category = "🗺️ 地图";
            return [p.file.link, category, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

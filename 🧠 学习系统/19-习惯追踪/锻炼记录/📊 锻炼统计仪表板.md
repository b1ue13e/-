## 📊 核心数据

```dataviewjs
const today = new Date().toISOString().split('T')[0];
const thisMonth = today.slice(0, 7);
const thisWeekStart = new Date();
thisWeekStart.setDate(thisWeekStart.getDate() - thisWeekStart.getDay() + 1);
const weekStartStr = thisWeekStart.toISOString().split('T')[0];

// 计算连续天数
const diaries = dv.pages('"🧠 学习系统/21-周期笔记/1-日记"').sort(p => p.file.name, 'desc');
let streak = 0;
for (let d of diaries) {
    if (d.锻炼打卡 === 1) {
        streak++;
    } else if (d.file.name !== today) {
        break;
    }
}

// 本月统计
const monthNotes = dv.pages('"🧠 学习系统/21-周期笔记/1-日记"').where(p => p.file.name.startsWith(thisMonth));
const monthWorkouts = monthNotes.where(p => p.锻炼打卡 === 1);
const monthCount = monthWorkouts.length;
const monthMinutes = monthWorkouts.map(p => p.运动时长 || 0).array().reduce((a, b) => a + b, 0);
const monthCalories = monthWorkouts.map(p => p.消耗热量 || 0).array().reduce((a, b) => a + b, 0);

// 本周统计
const weekNotes = dv.pages('"🧠 学习系统/21-周期笔记/1-日记"').where(p => p.file.name >= weekStartStr);
const weekWorkouts = weekNotes.where(p => p.锻炼打卡 === 1);
const weekCount = weekWorkouts.length;
const weekMinutes = weekWorkouts.map(p => p.运动时长 || 0).array().reduce((a, b) => a + b, 0);

dv.table(
    ["指标", "本周", "本月", "连续天数"],
    [
        ["运动次数", weekCount, monthCount, `${streak}天 🔥`],
        ["运动时长", `${weekMinutes}分钟`, `${Math.round(monthMinutes/60*10)/10}小时`, ""],
        ["消耗热量", "-", `${monthCalories}千卡`, ""],
    ]
);
```

## 📅 本周运动记录

```dataview
TABLE WITHOUT ID
    file.link AS "日期",
    choice(锻炼打卡 = 1, "✅", "❌") AS "完成",
    运动类型 AS "类型",
    运动时长 AS "时长",
    消耗热量 AS "热量",
    运动强度 AS "强度",
    运动感受 AS "感受"
FROM "🧠 学习系统/21-周期笔记/1-日记"
WHERE file.name >= substring(date(today) - dur(6 days), 0, 10)
SORT file.name ASC
```

## 📈 运动趋势（近30天）

const days = dv.pages('"🧠 学习系统/21-周期笔记/1-日记"')
    .sort(p => p.file.name, 'desc')
    .limit(30)
    .sort(p => p.file.name, 'asc');

const labels = [];

const durationData = [];

for (let d of days) {
    labels.push(d.file.name.slice(5));
    workoutData.push(d.锻炼打卡 === 1 ? 1 : 0);

dv.paragraph(`

   - title: 是否运动
     data: [${workoutData.join(',')}]
   - title: 运动时长(分)
     data: [${durationData.join(',')}]
\`\`\`
`);

## 🏃 运动类型统计

```dataviewjs
const thisMonth = new Date().toISOString().split('T')[0].slice(0, 7);
const workouts = dv.pages('"🧠 学习系统/21-周期笔记/1-日记"')
    .where(p => p.file.name.startsWith(thisMonth) && p.锻炼打卡 === 1 && p.运动类型);

const typeCount = {};
const typeDuration = {};

for (let w of workouts) {
    const type = w.运动类型 || "未记录";
    typeCount[type] = (typeCount[type] || 0) + 1;
    typeDuration[type] = (typeDuration[type] || 0) + (w.运动时长 || 0);
}

const data = Object.entries(typeCount).map(([type, count]) => [type, count, typeDuration[type]]);
dv.table(["运动类型", "次数", "总时长(分)"], data);
```

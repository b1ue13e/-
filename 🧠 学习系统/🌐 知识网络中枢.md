```dataviewjs
        const prereqNote = dv.pages('"🧠 学习系统"').where(p => p.topic === item.prereq && p.status === "completed");
        return prereqNote.length === 0;
    })
    .limit(10);

if (incompletePrereqs.length > 0) {
    dv.table(
        ["当前学习", "缺失前置知识", "建议先学"],
        incompletePrereqs.map(item => [
            item.note,
            item.prereq,
            `[[🔍 概念自动索引#${item.prereq}|查找笔记 →]]`
        ])
    );
} else {
    dv.paragraph("✅ 暂无前置知识缺口，可以继续深入学习！");
}
```

## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/05-计算机视觉"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let stage = "-";
            if (p.file.path.includes("01-")) stage = "🏗️ 基石";
            else if (p.file.path.includes("02-")) stage = "🧠 内核";
            else if (p.file.path.includes("03-")) stage = "🎯 任务";
            else if (p.file.path.includes("04-")) stage = "🚀 前沿";
            else if (p.file.path.includes("05-")) stage = "🏭 部署";
            return [p.file.link, stage, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

## 📊 学习统计

```dataviewjs
const normalizeStatus = status => {
    if (status === "进行中") return "ongoing";
    if (status === "已完成") return "completed";
    return status || "ongoing";
};
const getCreated = p => p.created || p.createTime || (p.file.ctime ? p.file.ctime.toFormat("yyyy-MM-dd") : "");
const notes = dv.pages('"🧠 学习系统/20-PARA知识管理/02-领域/学习笔记"')
    .where(p => p.file.name !== "📚 学习笔记目录")
    .array();
const totalNotes = notes.length;
const inProgress = notes.filter(p => normalizeStatus(p.status) === "ongoing").length;
const completed = notes.filter(p => normalizeStatus(p.status) === "completed").length;
const thisMonth = notes.filter(p => getCreated(p).startsWith(new Date().toISOString().slice(0, 7))).length;

dv.table(
    ["统计项", "数量"],
    [
        ["总笔记数", totalNotes],
        ["进行中", inProgress],
        ["已完成", completed],
        ["本月新增", thisMonth],
    ]
);
```

## 🎯 进行中的学习项目

```dataviewjs
const normalizeStatus = status => {
    if (status === "进行中") return "ongoing";
    if (status === "已完成") return "completed";
    return status || "ongoing";
};
const getCreated = p => p.created || p.createTime || "-";
const getUpdated = p => p.updated || p.updateTime || (p.file.mtime ? p.file.mtime.toFormat("yyyy-MM-dd") : "-");
const notes = dv.pages('"🧠 学习系统/20-PARA知识管理/02-领域/学习笔记"')
    .where(p => p.file.name !== "📚 学习笔记目录")
    .array()
    .filter(p => normalizeStatus(p.status) === "ongoing")
    .sort((a, b) => String(b.priority || "").localeCompare(String(a.priority || "")) || String(getCreated(b)).localeCompare(String(getCreated(a))));

dv.table(
    ["主题", "优先级", "难度", "开始时间", "更新时间"],
    notes.map(p => [p.file.link, p.priority || "-", p.difficulty || "-", getCreated(p), getUpdated(p)])
);
```

    if (status === "进行中") return "ongoing";
    if (status === "已完成") return "completed";

const getCreated = p => p.created || p.createTime || "-";

## 🏷️ 按标签分类

```dataviewjs
const notes = dv.pages('"🧠 学习系统/20-PARA知识管理/02-领域/学习笔记"');
const tagCount = {};

for (let n of notes) {
    if (n.tags) {
        for (let tag of n.tags) {
            if (tag !== "学习笔记") {
                tagCount[tag] = (tagCount[tag] || 0) + 1;
            }
        }
    }
}

const data = Object.entries(tagCount)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 15);

dv.table(["标签", "笔记数量"], data);
```

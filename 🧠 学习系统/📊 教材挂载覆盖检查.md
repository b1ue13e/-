# 📊 教材挂载覆盖检查

> 这页只做一件事：确认你的核心学术路径是否已经全部接入教材主线系统，以及哪些目录是刻意不做教材挂载的辅助目录。

```dataviewjs
const coreDirs = [
  "01-社交网络挖掘","02-时间序列分析","03-量化金融","04-离散数学","05-计算机视觉","06-常微分方程","07-最优化方法","08-NLP全栈","09-高等代数","10-数据结构与算法","11-计算机组成原理","12-信号与系统","13-强化学习","14-概率图模型","15-控制与机器人学","16-多元统计","17-机器学习",
  "22-概率论与数理统计","23-图神经网络","24-随机过程","25-测度论","26-实变函数","27-泛函分析","28-数学分析","29-运筹学","30-应用回归分析","31-属性数据分析","32-非参数统计","33-贝叶斯统计","34-偏微分方程","35-数值分析","36-数据挖掘与信息检索"
];

const covered = coreDirs.filter(dir =>
  app.vault.getAbstractFileByPath(`🧠 学习系统/${dir}/📚 教材挂载索引.md`)
);
const missing = coreDirs.filter(dir =>
  !app.vault.getAbstractFileByPath(`🧠 学习系统/${dir}/📚 教材挂载索引.md`)
);

dv.table(
  ["对象", "数量"],
  [
    ["核心学术路径总数", coreDirs.length],
    ["已接入教材主线", covered.length],
    ["未接入教材主线", missing.length],
    ["辅助目录（18-21）", 4]
  ]
);

const rate = coreDirs.length ? Math.round((covered.length / coreDirs.length) * 100) : 0;
dv.paragraph(`**核心学术路径覆盖率：${rate}%**`);
if (missing.length === 0) {
  dv.paragraph("**结论：01-17 与 22-36 的 32 条核心学术路径已全部接入教材主线系统。**");
} else {
  dv.paragraph("**仍未覆盖的核心路径：** " + missing.join("、"));
}
```

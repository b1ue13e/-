## 🎯 当前学习重点

```dataviewjs
dv.table(
    ["笔记", "阶段/维度", "更新日期", "状态"],
    dv.pages('"🧠 学习系统/36-数据挖掘与信息检索"')
        .where(p => !p.file.path.includes("🗺️"))
        .sort(p => p.file.mtime, 'desc')
        .limit(5)
        .map(p => {
            let category = "-";
            if (p.file.path.includes("01-")) category = "📘 挖掘";
            else if (p.file.path.includes("02-")) category = "📗 检索";
            else if (p.file.path.includes("03-")) category = "📙 系统";
            else if (p.file.path.includes("04-")) category = "📕 前沿";
            else if (p.file.path.includes("知识地图")) category = "🗺️ 地图";
            return [p.file.link, category, p.file.mtime.toFormat("MM-dd"), p.status || "ongoing"];
        })
);
```

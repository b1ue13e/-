## 📊 学习路径热区

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");
const notes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array();
const groups = {};
notes.forEach(p => {
    const folder = p.file.path.split('/')[1] || "未归类";
    if (!/^\d+-/.test(folder)) return;
    if (!groups[folder]) groups[folder] = { total: 0, ongoing: 0, planned: 0, latest: p.file.mtime };
    groups[folder].total += 1;
    if (p.status === "ongoing") groups[folder].ongoing += 1;
    if (["planned", "unseen", "locked"].includes(p.status)) groups[folder].planned += 1;
    if (p.file.mtime.ts > groups[folder].latest.ts) groups[folder].latest = p.file.mtime;
});

dv.table(
    ["路径", "笔记数", "进行中", "待启动", "最近更新"],
    Object.entries(groups)
        .sort((a, b) => b[1].latest.ts - a[1].latest.ts)
        .slice(0, 15)
        .map(([folder, item]) => [
            folder.replace(/^\d+-/, ""),
            item.total,
            item.ongoing,
            item.planned,
            item.latest.toFormat("yyyy-MM-dd")
        ])
);
```

```dataviewjs
        toList(p.concept).filter(isUsefulConcept).forEach(c => {
            const concept = normalizeConcept(c);
            if (!conceptPathMap[concept]) {
                conceptPathMap[concept] = new Set();
            }
            conceptPathMap[concept].add(path);
        });
    });

// 找出跨路径的概念
const crossPathConcepts = Object.entries(conceptPathMap)
    .filter(([concept, paths]) => paths.size > 1)
    .sort((a, b) => b[1].size - a[1].size)
    .slice(0, 5);

if (crossPathConcepts.length > 0) {
    dv.paragraph("🔗 **跨路径热门概念**：" +
        crossPathConcepts.map(([c, paths]) => `[[🔍 概念自动索引#${c}\|${c}]](${paths.size}路径)`).join(" · "));
}
```

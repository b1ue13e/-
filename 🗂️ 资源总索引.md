## 📊 系统健康看板

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name);
const isActualStagePage = p => p.type === "stage" && !p.template && inStudyScope(p);
const study = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).length;
const stages = dv.pages('"🧠 学习系统"').where(p => isActualStagePage(p)).length;
const activeProjects = dv.pages().where(p => p.type === "project" && !p.template && ["active", "ongoing"].includes(p.status)).length;
const projectsWithoutFocus = dv.pages().where(p => p.type === "project" && !p.template && ["active", "ongoing"].includes(p.status) && !p.focus).length;
const papers = dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).length;
const books = dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).length;
const missingType = dv.pages().where(p => !p.type && !p.template).length;

dv.table(
    ["对象", "数量", "治理提醒"],
    [
        ["学习笔记", study, "核心增量来自原子卡，不建议继续堆长目录"],
        ["阶段页", stages, "阶段页只做导航和验收，不替代学习卡"],
        ["活跃项目", activeProjects, projectsWithoutFocus > 0 ? `${projectsWithoutFocus} 个活跃项目缺 focus` : "focus 字段正常"],
        ["论文笔记", papers, "建议把关键概念同步进 concept"],
        ["读书笔记", books, "可迁移概念应回填到学习系统"],
        ["缺少 type 的页面", missingType, missingType > 0 ? "优先补 frontmatter" : "类型字段正常"]
    ]
);
```

## 📦 资源存量

```dataviewjs
const paperFiles = app.vault.getFiles().filter(f => f.path.startsWith("📚 论文库/PDF库/") || (f.path.startsWith("📚 论文库/") && f.extension?.toLowerCase() === "pdf")).length;
const bookFiles = app.vault.getFiles().filter(f =>
    f.path.startsWith("📚 书库/书籍资源/") &&
    ["pdf", "epub", "mobi", "azw3", "txt"].includes((f.extension || "").toLowerCase())
).length;
const markdownFiles = app.vault.getMarkdownFiles().length;

dv.table(
    ["资源类型", "数量", "入口"],
    [
        ["Markdown 页面", markdownFiles, "全库"],
        ["论文 PDF", paperFiles, dv.fileLink("📚 论文库/PDF库/📁 PDF索引", false, "PDF 索引")],
        ["书籍文件", bookFiles, dv.fileLink("📚 书库/📚 书籍主页", false, "书籍主页")]
    ]
);
```

## 📝 最近新增或更新

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const pages = [
    ...dv.pages('"🧠 学习系统"').where(p => ["study", "stage", "project"].includes(p.type) && !p.template && inStudyScope(p)).array(),
    ...dv.pages('"📚 论文库/论文笔记"').where(p => !p.template).array(),
    ...dv.pages('"📚 书库/书籍笔记"').where(p => !p.template).array()
];

dv.table(
    ["笔记", "类型", "状态", "日期"],
    pages
        .sort((a, b) => b.file.mtime.ts - a.file.mtime.ts)
        .slice(0, 18)
        .map(p => [p.file.link, p.type || "-", p.status || "-", p.file.mtime.toFormat("yyyy-MM-dd")])
);
```

const focusOrder = { now: 0, next: 1, later: 2 };
const toMillis = value => {
    if (!value) return Number.MAX_SAFE_INTEGER;
    if (value.ts) return value.ts;
    const parsed = new Date(value);
    return Number.isNaN(parsed.getTime()) ? Number.MAX_SAFE_INTEGER : parsed.getTime();
};
const activeProjects = dv.pages()

    .array()
    .sort((a, b) => {
        const focusDelta = (focusOrder[a.focus ?? "later"] ?? 9) - (focusOrder[b.focus ?? "later"] ?? 9);
        if (focusDelta !== 0) return focusDelta;
        return toMillis(a.deadline) - toMillis(b.deadline);

const excludedStudyDirs = [

const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isStageLikeName = name => /阶段|板块/.test(name || "");
const folderLabel = p => (p.file.folder?.split("/").pop() || "").replace(/^\d+-/, "").replace(/\s+/g, "");
const stageText = p => String(p.topic || p.file.name || "").replace(/\s+/g, "");

    if (!inStudyScope(p) || p.template) return false;

const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !isStageLikeName(p.file.name);
const studyNotes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).length;
const stagePages = dv.pages('"🧠 学习系统"').where(p => isActualStagePage(p)).length;

## 🧭 学习路径热区

```dataviewjs
const excludedStudyDirs = [
    "🧠 学习系统/18-题库学习/",
    "🧠 学习系统/19-习惯追踪/",
    "🧠 学习系统/20-PARA知识管理/",
    "🧠 学习系统/21-周期笔记/"
];
const inStudyScope = p => !excludedStudyDirs.some(dir => p.file.path.startsWith(dir));
const isAtomicStudyNote = p => p.type === "study" && !p.template && inStudyScope(p) && !/阶段|板块/.test(p.file.name || "");
const notes = dv.pages('"🧠 学习系统"').where(p => isAtomicStudyNote(p)).array();
const groups = {};
notes.forEach(p => {
    const folder = p.file.path.split('/')[1] || "未归类";
    if (!/^\d+-/.test(folder)) return;
    if (!groups[folder]) groups[folder] = { total: 0, ongoing: 0, latest: p.file.mtime };
    groups[folder].total += 1;
    if (p.status === "ongoing") groups[folder].ongoing += 1;
    if (p.file.mtime.ts > groups[folder].latest.ts) groups[folder].latest = p.file.mtime;
});

dv.table(
    ["路径", "笔记数", "进行中", "最近更新", "入口"],
    Object.entries(groups)
        .sort((a, b) => b[1].latest.ts - a[1].latest.ts)
        .slice(0, 12)
        .map(([folder, item]) => {
            const clean = folder.replace(/^\d+-/, "");
            return [clean, item.total, item.ongoing, item.latest.toFormat("yyyy-MM-dd"), `[[🧠 学习系统/${folder}/🗺️ ${clean}学习路径|进入]]`];
        })
);
```
